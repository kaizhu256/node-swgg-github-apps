swgg-github-apps (2018.2.24)

this zero-dependency package will provide a (nodejs-compatible) swagger-client for github-apps's web-apis, with a working web-demo

[download standalone app]

nodejs initialization

/*
 * initialize nodejs swgg-client
 * 1. download currently-loaded apis to file swagger.json:
 *     $ curl -L -o swagger.json "https://kaizhu256.github.io/node-swgg-github-apps/build..master..travis-ci.org/app/assets.swgg.swagger.json"
 * 2. npm install swgg
 *     $ npm install swgg
 * 3. run code below to initialize nodejs swgg-client
 * 4. (optional) edit file swagger.json to suit your needs
 */
var swgg;
swgg = require("swgg");
swgg.apiUpdate(require("./swagger.json"));
console.log("printing currently loaded apis ...");
Object.keys(swgg.apiDict).sort().forEach(function (key) {
    console.log("swgg.apiDict[" + JSON.stringify(key) + "].ajax");
});
console.log("initialized nodejs swgg-client");
